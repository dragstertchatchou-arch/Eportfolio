        // --- 1. GESTION DES ONGLETS (TABS) ---

        function handleRouting() {

            // Récupère le #mot dans l'URL (ex: #hobbies)

            const hash = window.location.hash.substring(1);

            

            // Si pas de hash, on affiche 'voyages' par défaut

            const activeId = hash || 'voyages';



            // On cache tout

            document.querySelectorAll('.tab-content').forEach(el => {

                el.classList.remove('active-tab');

            });



            // On affiche celui qui correspond

            const target = document.getElementById(activeId);

            if (target) {

                target.classList.add('active-tab');

            }

        }



        // Fonction pour changer manuellement (clic direct)

        function switchTab(tabId) {

            // Met à jour l'URL sans recharger

            history.pushState(null, null, '#' + tabId);

            handleRouting();

        }



        // Écouteurs d'événements (Chargement page + Changement URL)

        window.addEventListener('load', handleRouting);

        window.addEventListener('hashchange', handleRouting);





        // --- 2. GESTION DU SLIDER AUTOMATIQUE ---

        let slides = document.querySelectorAll('.slide');

        let currentSlide = 0;



        function nextSlide() {

            // Vérifie qu'on est sur l'onglet Voyages, sinon inutile de changer l'image

            if(document.getElementById('voyages').classList.contains('active-tab')) {

                slides[currentSlide].classList.remove('active');

                currentSlide = (currentSlide + 1) % slides.length;

                slides[currentSlide].classList.add('active');

            }

        }

        setInterval(nextSlide, 4000);
